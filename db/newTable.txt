CREATE TABLE movie (
  ID           int PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
  name         text,
  description  text,
  image        text,
  release_date date
);

CREATE TABLE reviewer (
  ID    int,
  name  varchar,
  Email varchar,
  PRIMARY KEY (ID)
);

CREATE TABLE movie_reviews (
  movieID     int,
  reviewerID  int,
  score       float,
  review      text,
  review_date date,
  PRIMARY KEY (movieID, reviewerID),
  FOREIGN KEY(movieID),
  FOREIGN KEY(reviewerID)
);
